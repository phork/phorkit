### Scroll to element

<Canvas>
  <Story name="Scroll to element" parameters={{ layout: 'fullscreen' }}>
    <ThemeWrapper>
      {({ themeId }) => (
        <ListRegistryProvider>
          <ListRegistryConsumer>
            {({ getItem, items }) => (
              <div style={{ padding: '10px' }}>
                <Looper
                  end={40}
                  render={i => (
                    <Rhythm key={i} m={1}>
                      <Button
                        color="primary"
                        onClick={() =>
                          getItem(i).current.scrollIntoView({ behavior: 'smooth', block: 'nearest', inline: 'start' })
                        }
                        shape="brick"
                        weight="shaded"
                      >{`Scroll to ${i}`}</Button>
                    </Rhythm>
                  )}
                  start={0}
                  step={10}
                />
                <div
                  className={cx(
                    scrollbar.scrollbar,
                    scrollbar['scrollbar--primary'],
                    scrollbar[`scrollbar--${themeId}`],
                  )}
                  style={{ marginTop: '12px', height: '100px', overflow: 'auto' }}
                >
                  <Looper
                    end={40}
                    render={i => (
                      <ListRegistryItem id={i} key={i}>
                        <Typography as="div" color="primary">
                          <Rhythm my={1} px={2}>{`Line number ${i}`}</Rhythm>
                        </Typography>
                      </ListRegistryItem>
                    )}
                    start={0}
                  />
                </div>
              </div>
            )}
          </ListRegistryConsumer>
        </ListRegistryProvider>
      )}
    </ThemeWrapper>
  </Story>
</Canvas>

### Scroll to element with function and re-render

<Canvas>
  <Story name="Scroll to element with function and re-render" parameters={{ layout: 'fullscreen' }}>
    <ThemeWrapper>
      {({ themeId }) => (
        <StateWrapper initialState={0}>
          {({ state: count, setState: setCount }) => (
            <ListRegistryProvider>
              <ListRegistryConsumer>
                {({ getItem }) => (
                  <div style={{ padding: '10px' }}>
                    <Looper
                      end={40}
                      render={i => (
                        <Rhythm key={i} m={1}>
                          <Button
                            color="primary"
                            onClick={() => {
                              setCount(count + 1);
                              getItem(i).current.scrollIntoView({
                                behavior: 'smooth',
                                block: 'nearest',
                                inline: 'start',
                              });
                            }}
                            shape="brick"
                            weight="shaded"
                          >{`Scroll to ${i}`}</Button>
                        </Rhythm>
                      )}
                      start={0}
                      step={10}
                    />
                    <div
                      className={cx(
                        scrollbar.scrollbar,
                        scrollbar['scrollbar--primary'],
                        scrollbar[`scrollbar--${themeId}`],
                      )}
                      style={{ marginTop: '12px', height: '100px', overflow: 'auto' }}
                    >
                      <Looper
                        end={40}
                        render={i => (
                          <ListRegistryItem id={i} key={i}>
                            <Typography as="div" color="primary">
                              <Rhythm my={1} px={2}>{`Line number ${i}${count > 0 ? ` (${count})` : ''}`}</Rhythm>
                            </Typography>
                          </ListRegistryItem>
                        )}
                        start={0}
                      />
                    </div>
                  </div>
                )}
              </ListRegistryConsumer>
            </ListRegistryProvider>
          )}
        </StateWrapper>
      )}
    </ThemeWrapper>
  </Story>
</Canvas>
