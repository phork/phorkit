#### Shapes

## Examples

<Canvas>
  <ThemeWrapper>
    <Rhythm grouped m={-2}>
      <Looper
        list={['solid', 'shaded', 'outlined', 'ghost']}
        render={weight => (
          <Rhythm grouped key={weight || 'undefined'} m={2}>
            <Looper
              list={['primary', 'success', 'warning', 'danger', 'neutral', 'black', 'white']}
              render={color => (
                <Rhythm key={`${weight}-${color}`} mx={2} my={1}>
                  <Button color={color} onClick={() => action(color)} weight={weight}>
                    {ucfirst(color)}
                  </Button>
                </Rhythm>
              )}
            />
            <br />
            <Looper
              list={['primary', 'success', 'warning', 'danger', 'neutral', 'black', 'white']}
              render={color => (
                <Rhythm key={`${weight}-${color}`} mx={2} my={1}>
                  <Button
                    disabled
                    loading
                    color={color}
                    loader={<SpinnerIcon scale="medium" />}
                    onClick={() => action(color)}
                    weight={weight}
                  >
                    {ucfirst(color)}
                  </Button>
                </Rhythm>
              )}
            />
          </Rhythm>
        )}
      />
    </Rhythm>
  </ThemeWrapper>
</Canvas>
