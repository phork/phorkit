## Full context

<Playground>
  <ThemeWrapper>
    <Banners style={{ marginTop: '18px' }}>
      <BannerConsumer>
        {({ createNotification, clearNotifications }) => (
          <Rhythm m={2}>
            <Button
              color="primary"
              onClick={() =>
                createNotification(
                  <Banner level="info" onClose={() => console.log('closed banner')}>
                    This is an info banner because something interesting happened. Exciting!
                  </Banner>,
                )
              }
            >
              Info banner
            </Button>
            <Button
              color="danger"
              onClick={() =>
                createNotification(
                  <Banner level="danger" onClose={() => console.log('closed banner')}>
                    This is a danger banner because something bad happened. Uh oh!
                  </Banner>,
                )
              }
            >
              Danger banner
            </Button>
            <Button
              color="warning"
              onClick={() =>
                createNotification(
                  <Banner permanent level="warning" onClose={() => console.log('closed banner')}>
                    This is a permanent warning banner because something not so great happened. Meh.
                  </Banner>,
                )
              }
            >
              Warning banner
            </Button>
            <Button
              color="success"
              onClick={() =>
                createNotification(
                  <Banner
                    contextId="demo-banner-success"
                    level="success"
                    onClose={(event, contextId) => console.log(`closed ${contextId}`)}
                  >
                    This is an overwriting success banner because it has got a contextId. Yay!
                    <div style={{ position: 'absolute', right: '8px', fontSize: '10px', opacity: 0.4 }}>
                      {Date.now()}
                    </div>
                  </Banner>,
                )
              }
            >
              Success banner
            </Button>
            <Button color="neutral" onClick={() => clearNotifications()}>
              Clear all
            </Button>
          </Rhythm>
        )}
      </BannerConsumer>
    </Banners>
  </ThemeWrapper>
</Playground>
