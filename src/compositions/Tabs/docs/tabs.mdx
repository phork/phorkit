### Custom tab styles, flush content

<Playground>
  <ThemeWrapper>
    <Tabs
      unstyled
      initialSelected={['second']}
      style={{ outline: 'none' }}
      onSelect={(event, id) => console.log(id)}
      items={[
        {
          id: 'first',
          label: ({ disabled, focused, selected }) => (
            <Rhythm mr={2}>
              <Button
                disabled={disabled}
                focused={focused}
                as="div"
                color="neutral"
                weight={selected ? undefined : 'outlined'}
              >
                First
              </Button>
            </Rhythm>
          ),
          content: ({ selected }) => <h4>First panel{selected ? ' is selected' : ''}</h4>,
          labelProps: { style: { outline: 'none' } },
          contentProps: { flush: true },
        },
        {
          id: 'second',
          label: ({ disabled, focused, selected }) => (
            <Rhythm mx={2}>
              <Button
                disabled={disabled}
                focused={focused}
                as="div"
                color="neutral"
                weight={selected ? undefined : 'outlined'}
              >
                Second
              </Button>
            </Rhythm>
          ),
          content: ({ selected }) => <h4>Second panel{selected ? ' is selected' : ''}</h4>,
          labelProps: { style: { outline: 'none' } },
          contentProps: { flush: true },
        },
        {
          id: 'third',
          label: ({ disabled, focused, selected }) => (
            <Rhythm mx={2}>
              <Button
                disabled={disabled}
                focused={focused}
                as="div"
                color="neutral"
                weight={selected ? undefined : 'outlined'}
              >
                Third
              </Button>
            </Rhythm>
          ),
          content: ({ selected }) => <h4>Third panel{selected ? ' is selected' : ''}</h4>,
          labelProps: { style: { outline: 'none', pointerEvents: 'none' } },
          contentProps: { flush: true },
          disabled: true,
        },
        {
          id: 'fourth',
          label: ({ disabled, focused, selected }) => (
            <Rhythm ml={2}>
              <Button
                disabled={disabled}
                focused={focused}
                as="div"
                color="neutral"
                weight={selected ? undefined : 'outlined'}
              >
                Fourth
              </Button>
            </Rhythm>
          ),
          content: ({ selected }) => <h4>Fourth panel{selected ? ' is selected' : ''}</h4>,
          labelProps: { style: { outline: 'none' } },
          contentProps: { flush: true },
        },
      ]}
    >
      {({ componentId, items, ref, themeId, vertical }) => (
        <TabsContainer style={{ border: 'none' }} vertical={vertical}>
          <TabList
            componentId={componentId}
            ref={ref}
            items={items}
            themeId={themeId}
            unstyled
            vertical={vertical}
            style={{
              alignItems: 'center',
              display: 'flex',
              flexDirection: vertical ? 'column' : 'row',
              outline: 'none',
            }}
          />
          <TabPanelGroup componentId={componentId} items={items} themeId={themeId} unstyled vertical={vertical} />
        </TabsContainer>
      )}
    </Tabs>
  </ThemeWrapper>
</Playground>
